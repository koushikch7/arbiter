vLLM Deployment Template (recommended) Docs ↗

Save as my_llm.py and run modal deploy my_llm.py

import modal

app = modal.App("llm-serve")
image = modal.Image.debian_slim().pip_install("vllm")

@app.cls(gpu="A10G", image=image, timeout=300)
class Model:
    @modal.enter()
    def load(self):
        from vllm import LLM
        self.llm = LLM(model="meta-llama/Llama-3.1-8B-Instruct")

    @modal.web_endpoint(method="POST")
    def serve(self, request: dict):
        from vllm import SamplingParams
        msgs = request.get("messages", [])
        prompt = " ".join(m["content"] for m in msgs)
        params = SamplingParams(
            temperature=request.get("temperature", 0.7),
            max_tokens=request.get("max_tokens", 512))
        output = self.llm.generate([prompt], params)[0]
        return {"choices": [{"message": {"role": "assistant",
                "content": output.outputs[0].text}, "finish_reason": "stop"}],
                "usage": {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}}
